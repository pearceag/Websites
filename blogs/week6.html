In this lab, I also learned how to convert a callback scheme into a promise scheme. I created a promise constructor ask() that expected the readline.question function as its parameter. Then, it hands that function another function, resolve which is called when my asynchronous operation finishes. You can see this implemented below: 

    const fsPromises = require('fs').promises;
    const path = require('path');
    const http = require('http');

    const readline = require('readline').createInterface({
        input: process.stdin,
        output: process.stdout
    });

    
    function ask() {
        return new Promise(resolve => {
            readline.question('> ', answer => resolve(answer));
        });
    }

    async function main() {
    while (true) {
        const response = await ask();
        switch(response) {
            case('exit'):
                console.log('Exiting...');
                process.exit();
                break;
            case('pwd'):
                console.log(process.cwd());
                break;
            default:
                break;
        }
    }
